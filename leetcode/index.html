vector<int> twoSum(vector<int> &num, int target) {
    unordered_map<int, int> numMap;

    int i = 0;
    for (; i < num.size(); i++) {
        numMap[num[i]] = i;
    }

    for (i = 0; i < num.size(); i++) {
        if (numMap.find(target - num[i]) != numMap.end() && numMap[target - num[i]] != i) {
            return {i, numMap[target - num[i]]};
        }
    }

    return {};
}

9. 回文数#

执行用时：12 ms, 在所有 C++ 提交中击败了82.79%的用户
内存消耗：6.1 MB, 在所有 C++ 提交中击败了17.67%的用户
class Solution {
public:
    bool isPalindrome(int x) {
        if (x < 0) {
            return false;
        } else {
            std::string a = std::to_string(x);

            int i = 0,j = a.size() - 1;

            while (i < j) {
                if (a[i++] != a[j--]) {
                    return false;
                }
            }

            return true;
        }
    }
};

26. 删除排序数组中的重复项#

执行用时：12 ms, 在所有 C++ 提交中击败了99.11%的用户
内存消耗：7.5 MB, 在所有 C++ 提交中击败了89.37%的用户
class Solution {
public:
    int removeDuplicates(vector<int>& nums) {
        int resize = 1, biao = 0;
        int len = nums.size();

        if (len == 0) return 0;
        if (len == 1) return 1;

        for (int i = 1; i < nums.size(); i++) {
            if (nums[i] == nums[biao]) {
                continue;
            }
            else {
                nums[resize] = nums[i];
                resize++;
                biao = i;
            }
        }
        return resize;
    }
};

35. 搜索插入位置#

二分查找

执行用时：8 ms, 在所有 C++ 提交中击败了35.92%的用户
内存消耗：6.5 MB, 在所有 C++ 提交中击败了92.21%的用户
int searchInsert(vector<int>& nums, int target) {
    int start = 0, end = nums.size() - 1, mid;

    if (nums.size() == 0) return 0;

    while (start <= end) {
        mid = (end + start) >> 1;

        if (nums[mid] == target) {
            return mid;
        } else if (nums[mid] < target) {
            start = mid + 1;
        } else {
            end = mid - 1;
        }

    }
    return start;
}
